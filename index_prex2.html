Creating your own 

const myObservable = new Observable(observer => {
    let counter = 0;
    let interval = setInterval(() => {
        observer.next( counter++ );
    }, 300);
    return () => { clearInterval(interval) }
});
myObservable.subscribe(next => {
    console.log(next) // 0, 1, 2, 3.....
})
